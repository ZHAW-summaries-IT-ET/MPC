\section{L3 -- OpenMP}

\begin{KR}{Recipe: Predict OpenMP Behaviour}\\
    \textbf{Read directives} \texttt{parallel} alone = whole block per thread (replicate); \texttt{parallel for} = iterations split.
    \textbf{Scope each var} \texttt{shared} (race risk) / \texttt{private} (uninit, discarded) / \texttt{firstprivate}/\texttt{lastprivate} / \texttt{reduction} (race-free).
    \textbf{Find races} shared var written by $>1$ iteration $\to$ indeterminate.
    \textbf{Count} multiply by thread count only in replicated regions.
\end{KR}

\begin{KR}{Recipe: Pick a Sync Primitive}\\
    one block at a time $\to$ \texttt{critical}; single memory op $\to$ \texttt{atomic}; accumulate $\to$ \texttt{reduction}; phase wall $\to$ barrier; cross-process $\to$ semaphore.
\end{KR}

\begin{KR}{Recipe: Choose a Schedule}\\
    even iterations $\to$ \texttt{static}; uneven $\to$ \texttt{dynamic}; tapering $\to$ \texttt{guided}.
\end{KR}

\begin{example2}{Ex.\ S7 -- parallel vs parallel for}\\
    With $T$ threads, $n=5$: how many prints for \texttt{parallel} alone vs \texttt{parallel for}?
    \tcblower
    \important{$T\times5$} (loop replicated) vs \important{$5$} (iterations shared). \texttt{parallel} makes threads; \texttt{for} splits work.
\end{example2}

\begin{example2}{Ex.\ S8 -- shared vs private}\\
    \texttt{shared(b)} loop then \texttt{private(b)} loop, both \texttt{b=a+i}. Final \texttt{b}?
    \tcblower
    S1 \texttt{shared}: \important{race} $\to$ any of $7..16$. S2 \texttt{private}: copies discarded $\to$ global \texttt{b} unchanged. Final = whatever S1 left.
\end{example2}

\begin{example2}{Ex.\ S18 -- Barrier internals}\\
    Spin or block on arrival? Who runs first on release?
    \tcblower
    \important{Depends}: short + spare cores $\to$ spin (\texttt{active}); long/oversubscribed $\to$ block (\texttt{passive}). Release order = \important{OS, nondeterministic}; use \texttt{ordered} if needed.
\end{example2}

\begin{example2}{Ex.\ S21 -- Reduction sum}\\
    How to sum an array race-free; what does \texttt{reduction} do?
    \tcblower
    \texttt{\#pragma omp parallel for reduction(+:sum)}: each thread keeps a \important{private partial}, combined after the join.
\end{example2}

\begin{example2}{Ex.\ S22 -- nowait}\\
    Effect of \texttt{nowait} on \texttt{single}?
    \tcblower
    \texttt{single} = one thread runs the block (implicit barrier). \texttt{nowait} \important{removes the barrier} $\to$ others proceed (safe only if no dependency).
\end{example2}

\begin{example2}{Ex.\ S28 -- Thread count}\\
    What determines the number of threads?
    \tcblower
    Precedence: \texttt{if} $\to$ \texttt{num\_threads} $\to$ \texttt{omp\_set\_num\_threads} $\to$ \texttt{OMP\_NUM\_THREADS} $\to$ default ($\approx$\#CPUs).
\end{example2}
